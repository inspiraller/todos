FROM mongo

ARG MONGO_DB_NAME
ARG MONGO_TABLE_COLLECTION_NAME

ENV ENV_MONGO_DB_NAME=$MONGO_DB_NAME
ENV ENV_MONGO_TABLE_COLLECTION_NAME=$MONGO_TABLE_COLLECTION_NAME

WORKDIR /scripts
COPY latest.csv .
COPY field_file_types.txt .
COPY script.sh .
RUN chmod +x script.sh
RUN echo "${ENV_MONGO_DB_NAME} ${ENV_MONGO_TABLE_COLLECTION_NAME}"
CMD /scripts/script.sh "${ENV_MONGO_DB_NAME}" "${ENV_MONGO_TABLE_COLLECTION_NAME}"
